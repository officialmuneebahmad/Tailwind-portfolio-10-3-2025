Determine alphabet or number Program # 1

To determine whether an input value is an alphabet or number 

              
  // Check if Character is an Alphabet or not

  #include <stdio.h>
  int main() {
    // 65 - 90 A - Z ----- ASCII CODES 
    // 97 - 122 a - z
    // 48 - 57 0 - 9 
    char a;
    printf("Please Enter a character : \n");
    scanf("%c",&a);
    
    if((a >= 65 && a <= 90) || (a >= 97 && a<= 122))   
    {
      printf("It's an alphabet! \n");
    }
    else if(a >= 48 && a <= 57){
      printf("It's a number! \n");
    }
    else{
      printf("Error! \n");
    }
    
    return 0;
  }

Prime numbers between two integers Program # 14

By using a loop we will print a series of prime numbers, the initial value and ending value is determined (or entered) by the user 

              
	// Prime Numbers Between two intervals
  #include 
  #include 

  int main() {
    int n1, n2, i, j, isPrime;

    printf("Enter starting number: ");
    scanf("%d", &n1);

    printf("Enter ending number: ");
    scanf("%d", &n2);

    if (n1 < 2) // To Check if n1(initial number) is less than 2. (We do not count 1 OK)
    
    n1 = 2;  // If YES start taking Prime numbers from 2
    printf("Prime numbers between %d and %d are:\n", n1, n2);
    
    for (i = n1; i <= n2; i++) {
        isPrime = 1;  // Assume i is prime. If i isPrime then apply condition below

        for (j = 2; j <= sqrt(i); j++) {		// <= Here i is all Prime numbers OK.
            if (i % j == 0) {
                isPrime = 0;  // If i is not prime then break the loop 
                break;
            }
        }

        if (isPrime) {  // If isPrime is true print all Prime numbers
            printf("%d ", i);
        }
    }

    printf("\n");
    
    getch();
    return 0;
}

Prime number or not Program # 15

Finding out that an entered value is a prime number or not , the logic is tricky but once figured out, it will be a piece of cake 

              
// Prime or Not

 #include 
 #include 
 int main() {
    int n, i, isPrime = 1;

    printf("Enter a number: ");
    scanf("%d", &n);

    if (n <= 1) {
        isPrime = 0;  // 0 and 1 are not prime
    } else {
        for (i = 2; i <= sqrt(n); i++) {
            if (n % i == 0) {
                isPrime = 0;
                break;
            }
        }
    }

    if (isPrime)
        printf("%d is a prime number.", n);
    else
        printf("%d is not a prime number.", n);

    return 0;
}

Building a calculator using switch Program # 17

A calculator can be built in many ways, a simple calculator is built using switch conditional statements 

              
// Simple Calculator Using switch...case

 #include 
 int main() {
	
	int n1, n2, op;
	
	printf("Enter first number : ");
	scanf("%d",&n1);
	
	printf("Enter second number : ");
	scanf("%d",&n2);
	
	printf("Choose Operation is be performed : \n");   
	printf("1) Division ( / ) \n");
	printf("2) Multiply ( x ) \n");
	printf("3) Addition ( + ) \n");
	printf("4) Subtraction ( - ) \n");
	
	scanf("%d",&op);
	
	switch (op){
		case 1 :
			printf("%d / %d = %d",n1,n2,n1/n2);
		break;
	
		case 2 :
			printf("%d x %d = %d",n1,n2,n1*n2);
		break;	
	
		case 3 :
			printf("%d + %d = %d",n1,n2,n1+n2);
		break;
	
		case 4 :
			printf("%d - %d = %d",n1,n2,n1-n2);
		break;
		
		default :
			printf("Error ");
	
	}
	getch();
	return 0;
}

Temperature conversion Program # 21

Temperature converter fehrenheit to celcius and vice versa , you only need is a formula 

              
  // Fahrenheit to Celsius vice versa

 #include 
 int main(){
	
	float temperature, F, C;
	int opt;
	printf("Choose Conversion : \n");
	printf("(1) Fahrenheit to Celsius \n");
	printf("(2) Celsius to Fahrenheit \n");
	scanf("%d",&opt);
	
	if(opt == 1){
		printf("Enter Fahrenheit Temperature : ");
		scanf("%f",&F);
		temperature = 0.556 * (F - 32);
		printf(" %.2f Fahrenheit into Celsius is %.2f C",F,temperature);
	}
	else if(opt == 2){
		printf("Enter Celsius Temperature : ");
		scanf("%f",&C);

		printf(" %.2f Celsius into Fahrenheit is %.2f F",C,temperature);
	}

		printf("Error");
	}
	return 0;
}
